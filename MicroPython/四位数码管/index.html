  • 四位数码管介绍

4 位 7 段数码管由一个 12 管脚的 4 位 7 段共阳极数码管和一个控制芯片 TM1650 构成。共阳 极数码管是一种显示器件，每个数码管由7个发光二极管(LED)组成，可以显示0到9的数字以及 一些字母和符号。TM1650是一个集成了数码管控制和驱动功能的芯片，它可以通过控制管脚来选 择要显示的数字，并提供了一些额外的功能，如亮度调节和扫描控制。这种组合可以广泛应用于各 种需要显示数字的设备，如计时器、温度计、电子秤等。

  • 第三方库

我们在libs文件夹中新建一个TFourDigitDisplay.py文件，把网上的地方放库保存进去就可以使用了：

# 导入MicroPython常量模块和I2C驱动
from micropython import const
from machine import I2C
import time

# 定义TM1650的命令地址（亮度控制等）和显示数据地址
COMMAND_I2C_ADDRESS = const(0x24)  # 命令模式I2C地址
DISPLAY_I2C_ADDRESS = const(0x34)  # 数据显示模式基地址（每位数码管地址递增）

# 数码管段码表（0-9, A-F的编码），对应共阴数码管段选信号
# 格式：g f e d c b a（7位二进制，高位到低位）
buf = (0x3F, 0x06, 0x5B, 0x4F, 0x66, 0x6D, 0x7D, 0x07, 0x7F, 0x6F, 0x77, 0x7C, 0x39, 0x5E, 0x79, 0x71)

class FourDigitDisplay():
    def __init__(self, i2c):
        """
        初始化数码管驱动
        :param i2c: 已配置的I2C对象（需提前设置SCL/SDA引脚和频率）
        """
        self.i2c = i2c            # 存储传入的I2C对象
        self._intensity = 3       # 默认亮度等级（0-8）
        self.dbuf = [0, 0, 0, 0]  # 数码管显示缓冲区，记录当前各段显示状态
        self.tbuf = bytearray(1) # 单字节缓冲区，用于I2C数据传输
        self.on()                 # 上电默认开启显示

    def intensity(self, dat=-1):
        """
        设置或获取亮度等级
        :param dat: 亮度值（0-8），若省略则返回当前亮度
        :return: 当前亮度值（当dat参数省略时）
        """
        if dat < 0 or dat > 8:    # 参数无效时返回当前亮度
            return self._intensity
        if dat == 0:              # 亮度为0时关闭显示
            self.off()
        else:
            self._intensity = dat
            self.cmd((dat << 4) | 0x01)  # 发送亮度命令（格式：0bLLLL1xxx）

    def cmd(self, c):
        """向命令地址发送控制命令（亮度/开关）"""
        self.tbuf[0] = c  # 将命令值存入缓冲区
        self.i2c.writeto(COMMAND_I2C_ADDRESS, self.tbuf)  # 发送到命令地址

    def dat(self, bit, d):
        """向指定数码管位发送显示数据"""
        self.tbuf[0] = d  # 将段码值存入缓冲区
        # DISPLAY_I2C_ADDRESS + bit%4 计算目标位的实际地址（0x34~0x37）
        self.i2c.writeto(DISPLAY_I2C_ADDRESS + bit % 4, self.tbuf)

    def on(self):
        """开启显示（使用当前亮度）"""
        self.cmd((self._intensity << 4) | 0x01)  # 命令格式：亮度级别 + 开启位

    def off(self):
        """关闭显示"""
        self._intensity = 0  # 关闭时重置亮度记录
        self.cmd(0)          # 发送全零命令（关闭显示）

    def clear(self):
        """清除所有数码管显示"""
        for i in range(4):   # 遍历4个数码管位
            self.dat(i, 0)   # 发送零段码（全灭）
        self.dbuf = [0, 0, 0, 0]  # 重置缓冲区

    def showbit(self, num, bit=0):
        """
        在指定位置显示单个数字
        :param num: 要显示的数字（0-15，超出取模）
        :param bit: 数码管位置（0-3，超出取模）
        """
        self.dbuf[bit % 4] = buf[num % 16]  # 更新缓冲区
        self.dat(bit, buf[num % 16])         # 发送段码数据

    def shownum(self, num):
        """
        显示十进制整数（支持负数，范围-999~9999）
        :param num: 要显示的整数（超出范围可能显示异常）
        """
        if num < 0:

            if -999 <= num < -100:
                self.dat(0, 0x40)
                self.showbit(abs(num) % 10, 3)                # 显示个位（第3位）
                self.showbit((abs(num) // 10) % 10, 2)        # 显示十位（第2位）
                self.showbit((abs(num) // 100) % 10, 1)      # 显示百位（第1位）
            elif -99 <= num < -10:
                self.dat(1, 0x40)
                self.showbit((abs(num) // 10) % 10, 2)        # 显示十位（第2位）
                self.showbit(abs(num) % 10, 3)                # 显示个位（第3位）
            else :
                self.dat(2, 0x40)
                self.showbit(abs(num) % 10, 3)                # 显示个位（第3位）

        else:
            str_num = "{:4d}".format(int(num))  # 格式化为4位字符串
            for i in range(4):
                if str_num[i] == " ":  # 空格表示需要关闭该位
                    self.hide_bit(i)
                else:
                    self.showbit(int(str_num[i]), i)


    def showhex(self, num):
        """
        显示十六进制数（支持负数，范围-0xFFFF~0xFFFF）
        :param num: 要显示的十六进制数
        """
        if num < 0:
            self.dat(0, 0x40)       # 显示负号
            num = -num
        else:
            self.showbit((num >> 12) % 16, 0)  # 显示最高4位（第0位）
        self.showbit(num % 16, 3)             # 显示最低4位（第3位）
        self.showbit((num >> 4) % 16, 2)      # 显示次低4位（第2位）
        self.showbit((num >> 8) % 16, 1)       # 显示中间4位（第1位）

    def showDP(self, bit=1, show=True):
        """
        控制小数点显示
        :param bit: 小数点位置（0-3），默认第1位（第二位显示）
        :param show: True显示小数点，False隐藏
        """
        bit = bit % 4  # 确保位置在0-3范围内
        if show:
            # 通过按位或操作添加小数点（段码最高位为小数点）
            self.dat(bit, self.dbuf[bit] | 0x80)
        else:
            # 通过按位与操作清除小数点
            self.dat(bit, self.dbuf[bit] & 0x7F)

# 一下是自己后加进去的方法       
    def hide_bit(self, bit):
        """关闭指定数码管位"""
        self.dat(bit, 0x00)  # 发送全零段码

    def show_dynamic(self, num):
        """智能显示（自动处理位数和小数点）"""
        if num < 0:
            self.dat(1, 0x40)  
#            self.showbit(0x40, 0)  # 显示负号
            num = abs(num)

        str_num = "{:4d}".format(int(num))  # 格式化为4位字符串
        for i in range(4):
            if str_num[i] == " ":  # 空格表示需要关闭该位
                self.hide_bit(i)
            else:
                self.showbit(int(str_num[i]), i)


    def countdown(self,start=9999,interval=0.5,val=-1):
        '''
        数字增/减方法
        start：起始数字
        interval：时间间隔，时间越短，数字变化越快
        val：数字跨度，+数为递增，-数为递减
        '''
        number = start
        while number >= 0:
            # 分解数字到各个位（千位 -> 个位）
            digits = [
                (number // 1000) % 10,  # 千位
                (number // 100) % 10,   # 百位
                (number // 10) % 10,    # 十位
                number % 10             # 个位
            ]

            # 计算实际有效位数（自动位数检测）
            if number == 0:
                valid_digits = 1
            else:
                valid_digits = len(str(abs(number)))

            # 更新显示（自动关闭前导零位）
            for i in range(4):
                if i < (4 - valid_digits):
                    self.hide_bit(i)     # 关闭无效的前导位
                else:
                    self.showbit(digits[i], i)  # 显示有效位

            time.sleep(interval)
            number += val

实验一：四位数码管的显示

程序如下：

'''
四位数码管的显示，不足4位前面补0
'''

from machine import I2C, Pin
import time
from libs.FourDigitDisplay import FourDigitDisplay  # 导入库

# 初始化I2C
i2c = I2C(scl=Pin(22), sda=Pin(21), freq=100000)
display = FourDigitDisplay(i2c)

'''
实验一、显示数字(-999~9999)
指定位置显示小数点
'''
display.shownum(1234)  # 显示数字1234
# display.showDP(bit=2, show=False)  # 在第三位显示小数点，show=Ture显示，False隐藏
time.sleep(2)

'''
实验二、在指定位置显示数字
num：显示的一个数字（0~9）
'''
display.clear() #清屏操作，避免之前的显示对后面的影响
display.showbit(num=1, bit=0)  # 在第一位上显示数字1
display.showbit(num=2, bit=1)  # 在第二位上显示数字2
display.showDP(bit=1, show=True ) # 注意：小数点的显示要放在数字显示的后面，否则会被覆盖
time.sleep(2)

实验二：四位数码管数字的变化

程序如下：

'''
实验二、四位数码管倒计时
不足4位前面不补0，显示空
start：起始数字
interval：时间间隔，时间越短，数字变化越快
val：数字跨度，+数为递增，-数为递减
'''
from machine import I2C, Pin
from libs.FourDigitDisplay import FourDigitDisplay  # 导入库:cite[2]

# 初始化I2C
i2c = I2C(scl=Pin(22), sda=Pin(21), freq=100000)
display = FourDigitDisplay(i2c)

# 启动倒计时（从9999开始）
display.countdown(start = 1002, interval = 1, val = -1)
