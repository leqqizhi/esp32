  • 矩阵键盘介绍

12键矩阵键盘含12个四脚按键，依靠金属弹片来保护受力情况，每个按键的四个引脚排列成矩阵的形式。在12键矩阵键盘模块中，当某个按钮被按压时，电路导通；当撤销这种压力的时候，电路断开。这个施压的力，就是用我们的手去开按钮、关按钮的动作。12键矩阵键盘共用一个模拟信号输入，当不同按键被按下时所输出的模拟值不同，由此可以判断是哪个按钮被按下。我们可以用一个GPIO口就可以控制12键矩阵键盘模块，可以大大减少需要使用IO接口。

实验一：解码矩阵键盘

程序如下

'''
解码矩阵键盘
'''
from machine import Pin,ADC
import time

# 在 32 引脚创建 ADC 对象
adc = ADC(Pin(32, Pin.IN),atten=ADC.ATTN_11DB)
while True:
    print(adc.read_u16())
    time.sleep(0.5)

注意事项：

  1. 将矩阵键盘平放，防止按压时抖动导致数值不稳定

  2. 按压时尽量保持用力稳定，防止抖动

实验四：校验密码（面向对象的实现方式）

  • 封装程序

对之前的函数进行封装，并保存到自建类文件目录common中，我这里起名叫DigitalKeypad

```python from machine import Pin, ADC import time

class DigitalKeypad: """数字键盘输入处理类，支持特殊功能键和输入管理

  属性：
      current_input (list): 当前输入缓存列表
      input_history (list): 历史输入记录存储列表
      latest_confirmed (list): 最新确认的输入列表
  """

  def __init__(self, adc_pin, buzzer_pin):
      # 初始化硬件组件
      self._adc = ADC(Pin(adc_pin, Pin.IN), atten=ADC.ATTN_11DB)
      self._buzzer = Pin(buzzer_pin, Pin.OUT)

      # 定义按键阈值表（下限，上限，显示标签）
      self._key_thresholds = [
          (0,      2000,  '1'),
          (2001,   3000,  '2'),
          (4001,   5000,  '3'),
          (6001,   7000,  '4'),
          (7501,   9000,  '5'),
          (9501,  10500,  '6'),
          (11001, 12000,  '7'),
          (12001, 13000,  '8'),
          (13001, 14000,  '9'),
          (14001, 15000,  '*'),
          (15501, 16500,  '0'),
          (17001, 19000,  '#')
      ]

      # 初始化输入状态
      self.current_input = []    # 当前输入缓存
      self.input_history = []    # 历史记录存储
      self.latest_confirmed = [] # 最新确认的输入
      self._is_inputting = False # 输入状态标志

  def _read_key(self):
      """读取当前按下的按键"""
      value = self._adc.read_u16()
      for low, high, char in self._key_thresholds:
          if low <= value <= high:
              return char
      return None

  def _beep(self, duration=0.1):
      """蜂鸣器提示音"""
      self._buzzer.on()
      time.sleep(duration)
      self._buzzer.off()

  def poll(self):
      """启动输入轮询监控（阻塞式）"""
      print("输入系统就绪（*清空/#确认）")
      while True:
          key = self._read_key()
          if key == '#':
              return self.confirm_input()
              break

          if key:
              self._beep()
              self._process_input(key)
          time.sleep(0.1)

  def _process_input(self, key):
      """处理输入逻辑"""
      if key == '*':
          self.current_input.clear()
          print("\n输入已重置")
      elif key == '#':
          return self.confirm_input()
      else:
          self.current_input.append(int(key))
          print(f"当前输入：{self.current_input}")

  def confirm_input(self):
      """确认当前输入并返回结果列表"""
      if not self.current_input:
          print("\n无输入内容")
          return None

      confirmed = self.current_input.copy()
      self.input_history.append(confirmed)
      self.latest_confirmed = confirmed
      self.current_input.clear()
      print(f"\n已确认输入：{confirmed}")
      return confirmed

# 使用示例 if name == "main": keypad = DigitalKeypad(32, 2)

  try:
      while True:
          # 非阻塞方式获取输入
          result = keypad.confirm_input()
          if result:
              print("获取到确认输入：", result)
              # 这里可以添加处理逻辑
          time.sleep(0.1)
  except KeyboardInterrupt:
      print("\n输入历史记录：", keypad.input_history)

```
