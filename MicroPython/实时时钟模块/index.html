  • 实时时钟模块

DS1307时钟模块是一款低功耗，具有56字节非失性RAM的全BCD码时钟日历实时时钟芯片，地址和数据通过两线双向的串行总线的传输，芯片可以提供秒，分，小时等信息，每一个月的天数能自动调整。AM/PM 标志位决定时钟工作于24小时或12小时模式，芯片有一个内置的电源感应电路，具有掉电检测和电池切换功能。DS1307时钟模块是一个集成了数字时钟的传感器，用户可以自己去编写程序，做成一个电子时钟。

  • 第三方库文件

我们在libs文件夹中新建一个ds1307.py。保存如下代码


from machine import I2C, Pin
import utime

class DS1307:
    def __init__(self, i2c_bus=0, sda_pin=21, scl_pin=22):
        self.i2c = I2C(i2c_bus, sda=Pin(sda_pin), scl=Pin(scl_pin))
        self.address = 0x68  # DS1307默认I2C地址

    def _bcd_to_dec(self, bcd):
        return (bcd // 16) * 10 + (bcd % 16)

    def _dec_to_bcd(self, dec):
        return (dec // 10) * 16 + (dec % 10)

    def set_time(self, year, month, day, hour, minute, second, weekday=0):
        data = bytearray(7)
        data[0] = self._dec_to_bcd(second) & 0x7F  # 秒（最高位为时钟停止位）
        data[1] = self._dec_to_bcd(minute)
        data[2] = self._dec_to_bcd(hour)  # 24小时制
        data[3] = self._dec_to_bcd(weekday + 1)  # DS1307的星期范围1-7
        data[4] = self._dec_to_bcd(day)
        data[5] = self._dec_to_bcd(month)
        data[6] = self._dec_to_bcd(year - 2000)  # 年份偏移量
        self.i2c.writeto_mem(self.address, 0x00, data)

    def get_time(self):
        data = self.i2c.readfrom_mem(self.address, 0x00, 7)
        return {
            'year': self._bcd_to_dec(data[6]) + 2000,
            'month': self._bcd_to_dec(data[5]),
            'day': self._bcd_to_dec(data[4]),
            'weekday': self._bcd_to_dec(data[3]) - 1,
            'hour': self._bcd_to_dec(data[2] & 0x3F),  # 屏蔽24/12小时标志位
            'minute': self._bcd_to_dec(data[1]),
            'second': self._bcd_to_dec(data[0] & 0x7F)  # 屏蔽时钟停止位
        }

    def enable_oscillator(self, enable=True):
        """控制时钟振荡器启停"""
        second_reg = self.i2c.readfrom_mem(self.address, 0x00, 1)[0]
        second_reg = second_reg & 0x7F if enable else second_reg | 0x80
        self.i2c.writeto_mem(self.address, 0x00, bytearray([second_reg]))

实验一：在OLED液晶屏现实实时

代码如下：

'''
实验1、实时显示时间
在OLED上实时显示时间
'''
# 导入库文件
from machine import Pin, RTC, I2C
import utime
from libs.ds1307 import DS1307
from libs.ssd1306 import SSD1306_I2C
import time

# ==========初始化OLED对象=================
i2c = I2C(0, scl=Pin(16), sda=Pin(17), freq=400000)  # ESP32常用引脚
oled = SSD1306_I2C(128, 64, i2c)  # 初始化对象

# ==========初始化DS1307实时时钟模块=================
rtc = DS1307(i2c_bus=1, sda_pin=21, scl_pin=22)

#  =======获取实时时间======
rtc1 =RTC()
datetime = rtc1.datetime()
print(datetime)

#=======初始化时间设置（从本地RTC中获取实时时间）===
rtc.set_time(datetime[0],datetime[1],datetime[2], datetime[4], datetime[5], datetime[6], datetime[3])

while True :
    current = rtc.get_time()
    print(current)
    weekday =f"{['Mon','Tue','Wed','Thu','Fri','Sat','Sun'][current['weekday']]}"
    str1 = f"{current['year']}-{current['month']:02d}-{current['day']:02d}"
    my_time = f"{current['hour']:02d}:{current['minute']:02d}:{current['second']:02d}"

    oled.text(str1, 0, 24)  # 在指定位置显示文本
    oled.text(weekday, 96, 24) 
    oled.text(my_time, 0, 40)
    oled.show()
    time.sleep(1)
    oled.fill(0)
