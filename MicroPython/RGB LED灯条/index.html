MicroPython NeoPixel 类详解

  • 参数说明：
参数 类型 说明 默认值
pin Pin对象 数据引脚 必填
n int LED数量 必填
bpp int 每像素字节数(3=RGB,4=RGBW) 3
timing int 时序模式(0=软件,1=硬件RMT) 0
  • 示例
# 导入库文件
from machine import Pin
from neopixel import NeoPixel

# 控制10个LED，GPIO2引脚
np = NeoPixel(Pin(2), 10)
  • 操作方法：
  1. 索引访问 (通过索引读写单个LED颜色)
# 设置第1个LED为红色
np[0] = (255, 0, 0)  

# 获取第2个LED颜色
color = np[1]  # 返回(R,G,B)元组

注意：

  • 修改后需调用write()生效

  • 索引从0开始，不能越界

  1. fill(color) 批量设置所有LED颜色）
# 全部设为蓝色
np.fill((0, 0, 255))

参数：

  • color: (R,G,B)或(R,G,B,W)元组
  1. write() 将颜色数据发送到LED灯带
np[0] = (255, 0, 0)
np.write()  # 实际更新LED

程序如下：

from machine import Pin,PWM
from neopixel import NeoPixel
import random


np = NeoPixel(Pin(2), 10)  # 控制10个LED，GPIO2
# 产生10个RGB列表值
list =[]
for i in range(10):
    list1 = []
    for j in range(3):
        list1.append(random.randint(0,255))
    list.append(list1)

for i in range(10):
    np[i] = list[i]
    np.write()

实验三：使用第三方库实现炫酷效果

  • WS2812B LED 简介

WS2812B 是一款智能控制 LED 光源，将控制电路和 RGB 芯片集成在 5050 封装中，形成完整像素点。

  • WS122812B库文件

MicroPython的开发中，我们会使用大量的库文件，可以直接拿来使用，大大提高了开发效率，降低了开发的难度。在项目文件夹中新建一个文件夹common（名字自己取，要求是英文）用来放置网上找来的库文件，然后在文件夹中，新建一个Python文件，我们示例为：WS2812B

import time
from machine import Pin
from neopixel import NeoPixel

# 流星雨动画效果
class WS2812B_1:
    '''
    流星雨流水灯效果
    :param pin: 灯条引脚
    :param num: 灯条中LED灯数量
    :param length: 流动亮条的长度，也就是同时发光的LED灯数量
    :param lolor: 灯光颜色，元组格式，元素数量为3，例如：color=(255, 0, 0)
    :param speed: 灯光流动的速度，单位：毫秒
    '''
    def __init__(self, pin, num,color=(255, 0, 0),length=5, speed=50):
        self.num = num
        self.pin = pin
        self.color = color
        self.length = length
        self.speed = speed
        self.np = NeoPixel(Pin(self.pin), self.num)

    def meteor_effect(self):
        for i in range(len(self.np) + self.length):
            for j in range(len(self.np)):
                # 计算亮度衰减
                brightness = max(0, 255 - (abs(i - j) * (255 // self.length)))
                self.np[j] = tuple(int(c * brightness / 255) for c in self.color)
            self.np.write()
            time.sleep_ms(self.speed)
            self.np.fill((0, 0, 0))



class WS2812B_2:
    '''
    全灯珠彩虹渐变
    :param pin: 灯条引脚
    :param num: 灯条中LED灯数量
    :param waith: 渐变的间隔时间
    '''
    def __init__(self, pin, num,waith=5):
        self.num = num
        self.pin = pin
        self.np = NeoPixel(Pin(self.pin), self.num)
        self.waith = waith
    def rainbow_cycle(self,wait=20):
        for j in range(256):  # 色相循环
            for i in range(len(self.np)):
                hue = (i * 256 // len(self.np)) + j
                self.np[i] = self.hsv_to_rgb(hue % 256, 255, 50)
            self.np.write()
            time.sleep_ms(self.waith)

    # HSV转RGB函数（简化版）
    def hsv_to_rgb(self,h, s=255, v=50):
        h = h % 256
        if s == 0:
            return (v, v, v)
        region = h // 43
        remainder = (h % 43) * 6
        p = (v * (255 - s)) >> 8
        q = (v * (255 - ((s * remainder) >> 8))) >> 8
        t = (v * (255 - ((s * (255 - remainder)) >> 8))) >> 8
        if region == 0:
            return (v, t, p)
        elif region == 1:
            return (q, v, p)
        elif region == 2:
            return (p, v, t)
        elif region == 3:
            return (p, q, v)
        elif region == 4:
            return (t, p, v)
        else:
            return (v, p, q)


if __name__ == "__main__":
    while True :
        led = WS2812B_1(pin =2,num=10, color = (0,255,255), length = 3, speed = 100)
        led.meteor_effect()

        led = WS2812B_2(pin =2,num=10,waith=5)
        led.rainbow_cycle()
