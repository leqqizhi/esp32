实验一、RGB三色显示

  • 程序如下：
'''
通过GPIO引脚控制红绿蓝三色的显示
'''
from machine import Pin,PWM
import time
R_p25 = Pin(25,Pin.OUT) # 红色对应的引脚
G_p26 = Pin(26,Pin.OUT) # 绿色对应的引脚
B_p27 = Pin(27,Pin.OUT) # 蓝色对应的引脚

while True:
    # 显示红色1秒后熄灭
    R_p25.value(1)  
    time.sleep(1)
    R_p25.value(0)
    time.sleep(1)

    # 显示绿色1秒后熄灭
    G_p26.value(1)
    time.sleep(1)
    G_p26.value(0)
    time.sleep(1)

    # 显示蓝色1秒后熄灭
    B_p27.value(1)
    time.sleep(1)
    B_p27.value(0)
    time.sleep(1)

    # 红绿蓝混色为白色显示1秒后熄灭
    R_p25.value(1)
    G_p26.value(1)
    B_p27.value(1)
    time.sleep(1)
    R_p25.value(0)
    G_p26.value(0)
    B_p27.value(0)
    time.sleep(1)

三基色混色法

实践证明，自然界中几乎所有的颜色都是可以由红色、绿色、蓝色按照不同的比率混合而成

三原色混色规律：

颜色组合 合成颜色 数学表示 十六进制示例 PWM值（0-255）
红(R) + 绿(G) 黄(Y) (255,255,0) #FFFF00 (255,255,0)
绿(G) + 蓝(B) 青(C) (0,255,255) #00FFFF (0,255,255)
红(R) + 蓝(B) 品红(M) (255,0,255) #FF00FF (255,0,255)
R+G+B 白(W) (255,255,255) #FFFFFF (255,255,255)
全关 黑(K) (0,0,0) #000000 (0,0,0)

RGB灯的工作原理：

RGB灯相当于在内置了3个LED灯，分别显示红、绿、蓝三种颜色，通过调节各个灯珠的亮度，从而达到不同的混色效果

实验三：随机产生一个颜色

程序如下

"""
每隔2秒随机组合一个颜色
"""
import time
from machine import Pin, PWM
import random

led_R =PWM(Pin(25),freq=1000,duty=0)
led_G =PWM(Pin(26),freq=1000,duty=0)
led_B =PWM(Pin(27),freq=1000,duty=0)

while True :
    led_R.duty(random.randint(0,1023))
    led_G.duty(random.randint(0,1023))
    led_B.duty(random.randint(0,1023))
    time.sleep(2)

实验四：更多的颜色变换

from machine import Pin, PWM
import time

# 初始化PWM引脚（根据实际接线修改）
red = PWM(Pin(25),freq=1000,duty=0)   # 红色引脚
green = PWM(Pin(26),freq=1000,duty=0) # 绿色引脚
blue = PWM(Pin(27),freq=1000,duty=0)  # 蓝色引脚

def set_color(r, g, b):
    """设置RGB颜色（0-255）"""
    # 将0-255映射到0-1023（ESP32 PWM范围）
    red.duty(int(r * 1023 / 255))
    green.duty(int(g * 1023 / 255))
    blue.duty(int(b * 1023 / 255))

# 基础颜色演示
colors = [
    (255, 0, 0),    # 红
    (0, 255, 0),    # 绿
    (0, 0, 255),    # 蓝
    (255, 255, 0),  # 黄
    (0, 255, 255),  # 青
    (255, 0, 255),  # 紫
    (255, 255, 255) # 白
]

set_color(0,0,255)

while True:
    for color in colors:
        print("显示颜色:", color)
        set_color(*color)
        time.sleep(1)  # 每种颜色显示1秒

    # 简单渐变效果（红→绿→蓝）
    for i in range(256):
        set_color(255-i, i, 0)  # 红→绿
        time.sleep_ms(10)
    for i in range(256):
        set_color(0, 255-i, i)  # 绿→蓝
        time.sleep_ms(10)
    for i in range(256):
        set_color(i, 0, 255-i)  # 蓝→红
        time.sleep_ms(10)

语法难点讲解

  • set_color(*color) * 符号用于 解包（unpack） 可迭代对象（如列表、元组等）。以下是详细讲解：

函数传参时的解包

def set_color(r, g, b):
    print(f"R:{r}, G:{g}, B:{b}")

color = (255, 128, 0)  # 元组包含三个值

# 正常调用
set_color(color[0], color[1], color[2])  # 传统写法
set_color(*color)  # 使用*解包 → 等效于 set_color(255, 128, 0)

案例中的程序

def set_color(r, g, b):
    print(f"R:{r}, G:{g}, B:{b}")

color = (255, 128, 0)  # 元组包含三个值

# 正常调用
set_color(color[0], color[1], color[2])  # 传统写法
set_color(*color)  # 使用*解包 → 等效于 set_color(255, 128, 0)

执行过程

  1. 第一次循环：color = (255, 0, 0)
    *color 解包 → set_color(255, 0, 0)

  2. 第二次循环：color = (0, 255, 0)
    *color 解包 → set_color(0, 255, 0)

  3. 第三次循环：color = (0, 0, 255)
    *color 解包 → set_color(0, 0, 255)

  4. 字典采用 ** 解包

params = {'r': 255, 'g': 0, 'b': 0}
set_color(**params)  # 等效于 set_color(r=255,g=0,b=0)

实验五：简化版3种颜色交替方案

from machine import Pin, PWM
import time

rgb = [PWM(Pin(p)) for p in [23,22,21]]  # R,G,B
[pwm.freq(1000) for pwm in rgb]          # 统一频率

def color(r,g,b):
    for i,v in enumerate([r,g,b]):
        rgb[i].duty(int(v * 1023 / 255))

# 红绿蓝交替显示
while True:
    color(255,0,0); time.sleep(1)  # 红
    color(0,255,0); time.sleep(1)  # 绿
    color(0,0,255); time.sleep(1)  # 蓝

enumerate()函数的核心用法

‌一、基本语法与参数说明‌

  • ‌函数定义‌：enumerate(iterable, start=0)
  • iterable：必需参数，接受列表、元组、字符串等可迭代对象。
  • start：可选参数，指定索引的起始值，默认为0。

将可迭代对象转换为枚举对象，每次迭代返回(index, value)元组

fruits = ['apple', 'banana', 'cherry']
for index, fruit in enumerate(fruits):
    print(f"索引{index}: {fruit}")
# 输出：
# 索引0: apple
# 索引1: banana
# 索引2: 

enum_list = list(enumerate(fruits))
print(enum_list)  # 输出：[(0, 'apple'), (1, 'banana'), (2, 'cherry')]
