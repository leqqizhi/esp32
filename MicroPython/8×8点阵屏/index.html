  • OLED显示屏介绍

8×8 点阵模块由 MAX7219 驱动，MAX7219 是一种集成化的串行输入/输出显示驱动器, 只需要 3个 IO 口即可驱动 1 个点阵，点阵显示时无闪烁且支持级联，8×8 点阵模块通过 Max7219 驱动，Max7219 采用的是三线串行通信，当接收 Arduino 数据时通过芯片内部 8×8 位静态 RAM 储存数据，在通过 Max7219 来控制 8×8 点阵模块的每个引脚的高低电平，从而达到显示目的。

  • 第三方库文件

我们在libs文件夹中新建一个MAX7219.py。保存如下代码

from machine import Pin, SPI,Timer
import time
from common.font import FONT

# matrix = MAX7219(spi_bus=2, cs_pin=5, num_devices=3)
class MAX7219:
    """
    MAX7219驱动类（适用于ESP32，支持级联）
    功能：控制8x8 LED点阵或数码管，支持亮度调节、像素操作、清屏等

    参数说明：
      - spi_bus: SPI总线编号（ESP32的HSPI=1, VSPI=2，默认VSPI=2）
      - cs_pin: 片选引脚（默认GPIO5）
      - num_devices: 级联的MAX7219数量（默认1个）
    """

    # MAX7219寄存器地址常量
    _REG_NOOP = 0x00
    _REG_DIGIT0 = 0x01  # 第1行数据（数码管则为第1位）
    _REG_DECODE = 0x09  # 解码模式
    _REG_INTENSITY = 0x0A  # 亮度
    _REG_SCAN_LIMIT = 0x0B  # 扫描行数
    _REG_SHUTDOWN = 0x0C    # 关机模式
    _REG_DISPLAY_TEST = 0x0F  # 测试模式

    def __init__(self, spi_bus=2, cs_pin=5, num_devices=1):
        """
        初始化SPI并配置MAX7219
        """
        # 初始化SPI（ESP32的VSPI默认引脚：sck=18, mosi=23）
        self.spi = SPI(
            spi_bus,
            baudrate=10_000_000,  # MAX7219支持最高10MHz时钟
            sck=Pin(18),
            mosi=Pin(23),
            polarity=0,  # SPI模式0（CPOL=0, CPHA=0）
            phase=0
        )

        # 初始化片选引脚
        self.cs = Pin(cs_pin, Pin.OUT)
        self.cs.value(1)  # 默认拉高（不选中）

        self.num_devices = num_devices  # 级联设备数量

        # 初始化MAX7219寄存器
        self._write_all(self._REG_SHUTDOWN, 0x01)     # 退出关机模式（0x01=正常）
        self._write_all(self._REG_DECODE, 0x00)       # 禁用BCD解码
        self._write_all(self._REG_SCAN_LIMIT, 0x07)   # 扫描所有8行
        self._write_all(self._REG_INTENSITY, 0x07)    # 默认亮度（0x00~0x0F）
        self._write_all(self._REG_DISPLAY_TEST, 0x00) # 关闭测试模式
        self.clear()  # 清空显示

    def _write(self, reg, data, device=0):
        """
        向指定设备写入数据（适用于级联场景）
        参数：
          - reg: 寄存器地址
          - data: 数据字节
          - device: 设备索引（从0开始，0=第一个模块）
        """
        if device >= self.num_devices:
            return

        # 级联时，数据按从后往前的顺序发送
        self.cs.value(0)
        for dev in reversed(range(self.num_devices)):
            if dev == device:
                self.spi.write(bytearray([reg, data]))
            else:
                self.spi.write(bytearray([self._REG_NOOP, 0x00]))
        self.cs.value(1)

    def _write_all(self, reg, data):
        """
        向所有级联设备写入相同数据（用于全局配置）
        """
        self.cs.value(0)
        for _ in range(self.num_devices):
            self.spi.write(bytearray([reg, data]))
        self.cs.value(1)

    def clear(self, device=None):
        """
        清空显示（可指定单个设备或全部）
        """
        if device is None:
            # 清空所有设备
            for row in range(8):
                self._write_all(self._REG_DIGIT0 + row, 0x00)
        else:
            # 清空指定设备
            for row in range(8):
                self._write(self._REG_DIGIT0 + row, 0x00, device)

    def set_pixel(self, x, y, state=1, device=0):
        """
        设置单个像素点状态（适用于点阵）
        参数：
          - x: 列坐标（0~7）
          - y: 行坐标（0~7）
          - state: 状态（1=亮，0=灭）
          - device: 设备索引
        """
        if not (0 <= x < 8 and 0 <= y < 8):
            return

        # 获取当前行数据
        current = self._get_row(y, device)
        # 修改对应位
        if state:
            new_data = current | (0x80 >> x)  # 高位对应左侧LED
        else:
            new_data = current & ~(0x80 >> x)
        self._write(self._REG_DIGIT0 + y, new_data, device)

    def _get_row(self, row, device):
        """
        读取指定行的当前数据（虚拟实现，实际需通过SPI读取）
        注意：MAX7219不支持直接读取显示数据，此处仅为演示逻辑
        """
        return 0x00  # 实际应用需维护一个显示缓冲区

    def set_brightness(self, brightness, device=None):
        """
        设置亮度（0-15）
        参数：
          - brightness: 亮度值
          - device: 指定设备（None=所有设备）
        """
        val = min(15, max(0, brightness))
        if device is None:
            self._write_all(self._REG_INTENSITY, val)
        else:
            self._write(self._REG_INTENSITY, val, device)

    def show_char(self, char_data, device=0):
        """
        显示8x8字符（需传入8字节的列表或字节序列）
        示例：CHAR_A = [0x3C, 0x7E, 0xC3, 0xC3, 0xFF, 0xC3, 0xC3, 0x00]
        """
        for row in range(8):
            self._write(self._REG_DIGIT0 + row, char_data[row], device)

    def show_str(self,text):
        '''
        显示字符串（需传入字符串，注意长度不要超过点阵屏幕数）
        '''
        i = 0
        for char in text:
            self.show_char(FONT[char],device=(i))
            i += 1

实验一：8×8点阵屏的显示

代码如下：

from machine import Pin, SPI
from libs.MAX7219 import MAX7219
import time
from common.font import FONT


# 初始化：使用VSPI（默认引脚），CS=GPIO5，级联1个模块(就是只有2个屏)
matrix = MAX7219(spi_bus=2, cs_pin=5, num_devices=3)

'''
1、 清屏方法 clear(device=None)
    device (int/None): 指定清空的设备索引（None表示清空所有设备）
    例如:清屏第二个屏幕：clear(1)
'''
matrix.clear()  # 清屏（所有）


'''
2、 设置像素点方法 set_pixel(x, y, state=1, device=0)
    x (int): 列坐标（0~7，0=最左侧）。
    y (int): 行坐标（0~7，0=最上方）。
    state (int): 状态（1=亮，0=灭）。
    device (int): 目标设备索引。(device是屏数的索引值，从0开始算第一个)
    示例：
    点亮第一个屏上的一个点，2秒后点亮第二个屏幕上的点，
    2秒后熄灭第一个点，再2秒熄灭第二个点

'''
matrix.set_pixel(3, 2, state=1, device=0)
time.sleep(2)
matrix.set_pixel(3, 2, state=1, device=1)
time.sleep(2)
matrix.clear(0)
time.sleep(2)
matrix.clear(device=1)
time.sleep(2)


'''
3、 设置亮度方法 set_brightness(brightness, device=None)
    brightness (int): 亮度值（0~15，0=最暗，15=最亮）。
    device (int/None): 指定设备索引（None=所有设备）
    示例：让一个点有暗到亮
'''
matrix.clear()  # 清屏
matrix.set_pixel(3, 2, state=1) # 点亮一个点
for i in range(16):
    matrix.set_brightness(i) # 亮度逐渐增加
    time.sleep(1)
matrix.clear()
