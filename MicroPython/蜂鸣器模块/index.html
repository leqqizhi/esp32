  • 蜂鸣器的介绍

蜂鸣器是一种一体化结构的电子讯响器。根据是否含有音源电路可以分为有源蜂鸣器和无源蜂鸣器。

  1. 有源蜂鸣器

    有源蜂鸣器是一种集成了驱动电路的蜂鸣器，它可以直接通过电流激励产生声音，不需要外部设备。有源蜂鸣器内部集成了振片、驱动电路和共振腔。当给有源蜂鸣器提供电流时，它会根据电流的变化产生声音。我们可以通过控制电流的大小和频率来控制蜂鸣器的声音。一般来说，我们可以通过改变输入电流的大小来调整音量，通过改变输入电流的频率来调整音调和音乐效果。

  2. 无源蜂鸣器

    无源蜂鸣器是一种简单的声音发生器，它通常由振片和共振腔组成。无源蜂鸣器不具备驱动电路，因此需要外部的电子设备来产生声音。当给无源蜂鸣器施加交变电压时，振片会振动并产生声音。无源蜂鸣器的工作频率由施加的电压频率决定。在使用无源蜂鸣器时，我们需要通过控制电压的频率和占空比来控制蜂鸣器的声音。通过改变交变电压的频率和占空比，我们可以产生不同的音调和音乐效果。

实验一：有源蜂鸣器的鸣响

'''
实验一：有源蜂鸣器鸣响
'''
from machine import Pin
import time

# 初始化蜂鸣器（假设接GPIO23）
buzzer = Pin(2, Pin.OUT)

'''
示例1：单次鸣响
'''
buzzer.on()      # 开启蜂鸣器
time.sleep(1)
buzzer.off()     # 关闭蜂鸣器

'''
示例2：间歇鸣响（滴滴声）
'''
def beep_intermittent(times=3, on_time=0.1, off_time=0.1):
    '''
    times：控制鸣响的次数
    on_time：通电时间间隔
    off_time：断电时间间隔
    '''
    """间歇鸣响模式"""
    for _ in range(times):
        buzzer.on()
        time.sleep(on_time)
        buzzer.off()
        time.sleep(off_time)

# 示例：快速鸣响3次（每次0.1响，0.1秒停）
beep_intermittent(times=3, on_time=0.1, off_time=0.1)

实验二：无源蜂鸣器的鸣响

'''
实验2、无源蜂鸣器音量的控制（响3秒后停止）
'''
from machine import PWM,Pin
import time

# 初始化PWM（频率1kHz）
pwm = PWM(Pin(2), freq=1000, duty=0)  # duty范围0-1023

def beep_pwm(duration=1, volume=512):
    """通过PWM调节音量"""
    pwm.duty(volume)  # 音量设置（0静音，1023最大）
    time.sleep(duration)
    pwm.duty(0)

# 示例：50%音量鸣响1秒
beep_pwm(duration=1, volume=512)
time.sleep(3)
beep_pwm(duration=1, volume=0)

'''
实验3：控制音调和音量
'''
from machine import Pin, PWM
import time

# 初始化蜂鸣器（支持任意PWM引脚）
buzzer = PWM(Pin(2), freq=1000, duty=0)  # 初始静音

def play_tone(freq, duration=0.5, volume=512):
    '''
    freq：发生频率（也就是音调）
    duration：发生持续时间
    volume：控制音量
    '''
    buzzer.freq(freq)        # 设置频率（单位：Hz）
    buzzer.duty(volume)      # 音量（0静音，1023最大）
    time.sleep(duration)
    buzzer.duty(0)           # 停止发声

# 示例：播放中音C（频率262Hz），持续1秒
play_tone(262, duration=1)

'''
实验4：测试各音调
'''
NOTE_C4 = 262   # 中音C
NOTE_D4 = 294   # D
NOTE_E4 = 330   # E
NOTE_F4 = 349   # F
NOTE_G4 = 392   # G
NOTE_A4 = 440   # A
NOTE_B4 = 494   # B

play_tone(NOTE_C4, duration=1)
play_tone(NOTE_D4, duration=1)
play_tone(NOTE_E4, duration=1)
play_tone(NOTE_F4, duration=1)
play_tone(NOTE_G4, duration=1)
play_tone(NOTE_A4, duration=1)
play_tone(NOTE_B4, duration=1)

'''
实验4：生日歌
'''
birthday_notes = [
    (NOTE_C4, 0.5), (NOTE_C4, 0.5), (NOTE_D4, 1), (NOTE_C4, 1), (NOTE_F4, 1),
    (NOTE_E4, 2), (NOTE_C4, 0.5), (NOTE_C4, 0.5), (NOTE_D4, 1), (NOTE_C4, 1),
    (NOTE_G4, 1), (NOTE_F4, 2)
]

def play_melody(notes, tempo=1):
    """播放旋律"""
    for note, duration in notes:
        play_tone(note, duration * tempo)  # tempo控制整体速度
        time.sleep(0.05)  # 音符间隔

# 播放示例
play_melody(birthday_notes)

实验三：播放小星星（面向对象）

把之前的程序进行整理，并封装到common文件下。

from machine import Pin, PWM
import time

class PassiveBuzzer:
    '''
    无源蜂鸣器发声
    :param  pins: pin对应引脚
    :param  notes是对应乐谱
    '''
    def __init__(self, pin):
        self.pwm = PWM(Pin(pin), freq=1000, duty=0)

    def play(self, freq, duration=0.5, volume=512):
        self.pwm.freq(freq)
        self.pwm.duty(volume)
        time.sleep(duration)
        self.pwm.duty(0)

    def stop(self):
        self.pwm.duty(0)

    def melody(self, notes, tempo=1.0):
        for note, beats in notes:
            if note == 0:  # 休止符
                time.sleep(beats * tempo)
            else:
                self.play(note, beats * tempo)
                time.sleep(0.05)  # 音符间隔
