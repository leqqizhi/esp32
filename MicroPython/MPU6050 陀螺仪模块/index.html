  • 红外遥控模块

MPU6050陀螺仪模块是一款高性能三轴加速度+三轴陀螺仪的六轴传感器模块，采用MPU6050芯片作为核心，MPU-6050是6轴运动处理传感器。它集成了3轴MEMS陀螺仪，3轴MEMS加速度计，以及一个可扩展的数字运动处理器DMP。使用它就是为了得到待测物体（如四轴飞行器、平衡小车）x、y、z轴的倾角（俯仰角Pitch、翻滚角Roll、偏航角Yaw）。我们通过I2C读取MPU6050的六个数据（三轴加速度AD值、三轴角速度AD值）经过姿态融合后就可以得到Pitch、Roll、Yaw角。利用自带的数字运动处理器DMP，通过IIC接口，输出姿态解算后的数据。具有体积小，使用方便等特点。模块自带2个定位孔，方便你将模块固定在其他设备。

  • 第三方库文件

我们在common文件夹中新建一个mpu6050.py。保存如下代码


import machine
import time
import math

# 定义加速度计/陀螺仪传感器操作类
class accel():
    def __init__(self, scl, sda ,addr=0x68):
        """
        初始化传感器
        :param i2c: 已初始化的I2C总线对象
        :param addr: 传感器I2C地址（MPU6050默认为0x68）
        """
        self.iic = machine.SoftI2C(scl = machine.Pin(scl), sda = machine.Pin(sda), freq = 100000)       # 存储I2C总线对象
        self.addr = addr     # 存储设备地址

        # 初始化传感器（退出睡眠模式）
        self.iic.start()     # 启动I2C通信
        # 向电源管理寄存器（0x6B）写入0唤醒设备
        # 107是寄存器地址的十进制表示（0x6B的十进制是107）
        # bytearray([107, 0])表示：向寄存器0x6B写入0x00
        self.iic.writeto(self.addr, bytearray([107, 0]))
        self.iic.stop()      # 结束I2C通信


    def get_raw_values(self):
        """ 获取原始字节数据（14字节） """
        self.iic.start()
        # 从寄存器0x3B开始连续读取14字节数据：
        # 加速度计X/Y/Z（各2字节） + 温度（2字节） + 陀螺仪X/Y/Z（各2字节）
        a = self.iic.readfrom_mem(self.addr, 0x3B, 14)
        self.iic.stop()
        return a  # 返回字节数组格式的原始数据

    def get_ints(self):
        """ 将原始字节转换为整数列表（调试用） """
        b = self.get_raw_values()
        c = []
        for i in b:
            c.append(i)  # 将字节逐个转换为0-255的整数
        return c  # 示例返回：[255, 128, ...]（14个元素）

    def bytes_toint(self, firstbyte, secondbyte):
        """ 
        将两个字节转换为有符号整数 
        （处理二进制补码格式）
        :param firstbyte: 高位字节
        :param secondbyte: 低位字节
        :return: 有符号整数（范围-32768~32767）
        """
        # 判断是否为负数（最高位为1）
        if not firstbyte & 0x80:
            # 正数直接拼接
            return (firstbyte << 8) | secondbyte
        else:
            # 负数处理（补码转换）
            # 1. 按位取反高位字节（^255等价于~操作）
            # 2. 按位取反低位字节并+1
            # 3. 组合结果后取负数
            return - (((firstbyte ^ 255) << 8) | (secondbyte ^ 255) + 1)

    def get_values(self):
        """ 获取解析后的传感器数值字典 """
        raw_ints = self.get_raw_values()  # 获取原始字节数据
        vals = {}  # 创建存储解析值的字典

        # 解析加速度计数据（X/Y/Z轴）
        # 每个轴占用2个字节（高位在前）
        vals["AcX"] = self.bytes_toint(raw_ints[0], raw_ints[1])  # 加速度X轴
        vals["AcY"] = self.bytes_toint(raw_ints[2], raw_ints[3])  # 加速度Y轴
        vals["AcZ"] = self.bytes_toint(raw_ints[4], raw_ints[5])  # 加速度Z轴

        # 解析温度数据（公式来自MPU6050手册）
        # 原始值范围：-40°C ~ +85°C
        vals["Tmp"] = self.bytes_toint(raw_ints[6], raw_ints[7]) / 340.00 + 36.53

        # 解析陀螺仪数据（X/Y/Z轴）
        vals["GyX"] = self.bytes_toint(raw_ints[8], raw_ints[9])   # 陀螺仪X轴
        vals["GyY"] = self.bytes_toint(raw_ints[10], raw_ints[11]) # 陀螺仪Y轴
        vals["GyZ"] = self.bytes_toint(raw_ints[12], raw_ints[13]) # 陀螺仪Z轴

        return vals  # 返回字典格式的原始整数值
        # 注意：返回的是未校准的原始数据！
        # 加速度计单位：LSB（需转换为g，默认±2g量程时 16384 LSB/g）
        # 陀螺仪单位：LSB（需转换为°/s，默认±250°/s量程时 131 LSB/°/s）
        # 温度单位：摄氏度

    def val_test(self):
        """ 
        持续测试传感器数据（仅用于测试！）
        警告：高频读取可能导致I2C总线崩溃
        """
        from time import sleep
        while True:
            print(self.get_values())  # 打印当前传感器值
            sleep(0.05)  # 50ms间隔（约20Hz采样率）



        # 加速度计计算角度（快速响应但受运动影响）
        pitch_acc = math.atan2(ay, math.sqrt(ax**2 + az**2)) * 180/math.pi
        roll_acc = math.atan2(-ax, math.sqrt(ay**2 + az**2)) * 180/math.pi

        # 陀螺仪积分角度（长期稳定但会漂移）
        pitch_gyro = self.pitch + gx * dt
        roll_gyro = self.roll + gy * dt

        # 互补滤波融合（权重系数alpha）
        alpha = 0.02  # 加速度计权重2%
        self.pitch = (1-alpha)*pitch_gyro + alpha*pitch_acc
        self.roll = (1-alpha)*roll_gyro + alpha*roll_acc

        # 角度归一化
        self.pitch = self._normalize_angle(self.pitch)
        self.roll = self._normalize_angle(self.roll)

        return (self.pitch, self.roll)

    def reset_angles(self):
        """ 重置姿态角归零（用于重新定位基准） """
        self.pitch = 0.0
        self.roll = 0.0

实验一：读取陀螺仪的数值

代码如下：

import machine
import time
import math

from libs.mpu6050 import accel

#  初始化对象
accel = accel(scl=22,sda=21)
while True:
    temperature = accel.get_values()['Tmp']   # 获取温度值
    Acx = accel.get_values()['AcX']     # X轴加速度
    Acy = accel.get_values()['AcY']     # Y轴加速度
    Acz = accel.get_values()['AcZ']     # Y轴加速度

    gyx = accel.get_values()['GyX']     # X轴角加速度
    gyy = accel.get_values()['GyY']     # Y轴角加速度
    gyz = accel.get_values()['GyZ']     # Z轴角加速度

#     print(accel.get_values())   # 原始数据
    print("温度: %.1f℃" % temperature) 
    print(f"X轴加速度：%d\t" % Acx, end='')
    print(f"Y轴加速度：%d\t" % Acy, end='')
    print(f"Z轴加速度：%d" % Acz)
    print(f"X轴角加速度：%d\t" % gyx, end='')
    print(f"Y轴角加速度：%d\t" % gyy, end='')
    print(f"Z轴角加速度：%d\t" % gyz)
    print('-'*45)
    time.sleep_ms(300)
